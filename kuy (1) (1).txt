-- ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏•‡πâ‡∏≤‡∏á ESP
function ESP:Clear()
    for player, data in pairs(self.Objects) do
        if data.Highlight then
            data.Highlight:Destroy()
        end
    end
    self.Objects = {}
    
    for _, conn in pairs(self.Connections) do
        if conn then conn:Disconnect() end
    end
    self.Connections = {}
    
    if self.LoopConnection then
        self.LoopConnection:Disconnect()
        self.LoopConnection = nil
    end
end

-- ‡∏™‡∏£‡πâ‡∏≤‡∏á ESP ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô
function ESP:Create(player)
    if player == LocalPlayer then return end
    if not player.Character then return end
    if self.Objects[player] then return end
    
    local char = player.Character
    if not char:FindFirstChild("Humanoid") then return end
    
    local hl = Instance.new("Highlight")
    hl.Name = "XESPHighlight"
    hl.Adornee = char
    hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    hl.FillColor = player.Team and player.Team.TeamColor.Color or Color3.fromRGB(255, 80, 80)
    hl.FillTransparency = self.Transparency
    hl.OutlineColor = Color3.fromRGB(255, 255, 255)
    hl.OutlineTransparency = 0.2
    
    -- ‡πÉ‡∏™‡πà‡πÉ‡∏ô CoreGui ‡∏õ‡πâ‡∏≠‡∏á‡∏Å‡∏±‡∏ô‡∏´‡∏≤‡∏¢ (‡πÉ‡∏ä‡πâ pcall ‡∏õ‡πâ‡∏≠‡∏á‡∏Å‡∏±‡∏ô error)
    pcall(function()
        hl.Parent = game:GetService("CoreGui")
    end)
    if not hl.Parent then
        hl.Parent = workspace.CurrentCamera
    end
    
    self.Objects[player] = {
        Highlight = hl,
        Character = char
    }
end

-- ‡∏≠‡∏±‡∏û‡πÄ‡∏î‡∏ó ESP
function ESP:Update()
    if not self.Enabled then return end
    
    -- ‡∏™‡∏£‡πâ‡∏≤‡∏á ESP ‡πÉ‡∏´‡πâ‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô‡∏ó‡∏µ‡πà‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡∏°‡∏µ
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            if not self.Objects[player] or self.Objects[player].Character ~= player.Character then
                self:Create(player)
            end
        end
    end
    
    -- ‡∏•‡∏ö ESP ‡∏Ç‡∏≠‡∏á‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô‡∏ó‡∏µ‡πà‡∏≠‡∏≠‡∏Å‡πÑ‡∏õ‡πÅ‡∏•‡πâ‡∏ß
    for player, data in pairs(self.Objects) do
        if not player.Parent or not data.Highlight or not data.Highlight.Parent then
            if data.Highlight then data.Highlight:Destroy() end
            self.Objects[player] = nil
        end
    end
end

-- ‡πÄ‡∏õ‡∏¥‡∏î/‡∏õ‡∏¥‡∏î ESP
function ESP:Toggle()
    self.Enabled = not self.Enabled
    
    if self.Enabled then
        self:Clear()
        print("[ESP] ‡πÄ‡∏õ‡∏¥‡∏î‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô")
        
        -- ‡∏™‡∏£‡πâ‡∏≤‡∏á‡πÉ‡∏´‡πâ‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô‡∏õ‡∏±‡∏à‡∏à‡∏∏‡∏ö‡∏±‡∏ô‡∏ó‡∏±‡∏ô‡∏ó‡∏µ
        for _, player in ipairs(Players:GetPlayers()) do
            self:Create(player)
        end
        
        -- Loop ‡∏≠‡∏±‡∏û‡πÄ‡∏î‡∏ó
        self.LoopConnection = game:GetService("RunService").Heartbeat:Connect(function()
            if tick() - self.LastUpdate >= 0.1 then
                self.LastUpdate = tick()
                self:Update()
            end
        end)
        
        -- Event ‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô‡πÉ‡∏´‡∏°‡πà
        self.Connections.PlayerAdded = Players.PlayerAdded:Connect(function(player)
            task.delay(1, function()
                if self.Enabled then
                    self:Create(player)
                end
            end)
        end)
        
    else
        print("[ESP] ‡∏õ‡∏¥‡∏î‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô")
        self:Clear()
    end
    
    return self.Enabled
end

}

function ItemESP:IsItem(obj)
    if not obj or not obj:IsA("Model") then return false end
    if obj.Parent == nil then return false end
    
    -- ‡πÄ‡∏ä‡πá‡∏Ñ Cache ‡∏Å‡πà‡∏≠‡∏ô
    if self.Cache[obj] ~= nil then
        return self.Cache[obj]
    end
    
    -- ‡πÑ‡∏°‡πà‡πÉ‡∏ä‡πà‡∏ï‡∏±‡∏ß‡∏•‡∏∞‡∏Ñ‡∏£‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô
    if game.Players:GetPlayerFromCharacter(obj) then
        self.Cache[obj] = false
        return false
    end
    
    -- ‡∏ï‡πâ‡∏≠‡∏á‡∏°‡∏µ PrimaryPart
    local primary = obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart")
    if not primary then
        self.Cache[obj] = false
        return false
    end
    
    -- ‡∏Ç‡∏ô‡∏≤‡∏î‡πÑ‡∏°‡πà‡πÉ‡∏´‡∏ç‡πà‡πÄ‡∏Å‡∏¥‡∏ô‡πÑ‡∏õ
    local size = obj:GetExtentsSize()
    if size.Magnitude > 20 then
        self.Cache[obj] = false
        return false
    end
    
    -- ‡πÄ‡∏ä‡πá‡∏Ñ‡∏ä‡∏∑‡πà‡∏≠‡πÑ‡∏≠‡πÄ‡∏ó‡πá‡∏°
    local name = obj.Name:lower()
    local itemKeywords = {"item", "coin", "gem", "weapon", "tool", "potion", "food", "material", "chest", "box", "loot", "collect", "reward", "cash", "money", "gun", "sword"}
    for _, keyword in ipairs(itemKeywords) do
        if name:find(keyword) then
            self.Cache[obj] = true
            return true
        end
    end
    
    -- ‡πÄ‡∏ä‡πá‡∏Ñ Mesh/Texture
    local hasVisual = obj:FindFirstChildWhichIsA("SpecialMesh") or 
                      obj:FindFirstChildWhichIsA("Texture") or
                      obj:FindFirstChildWhichIsA("Decal") or
                      primary:IsA("MeshPart")
    
    if hasVisual then
        self.Cache[obj] = true
        return true
    end
    
    self.Cache[obj] = false
    return false
end

function ItemESP:Clear()
    for _, data in pairs(self.Objects) do
        if data.Highlight then data.Highlight:Destroy() end
        if data.Billboard then data.Billboard:Destroy() end
    end
    self.Objects = {}
    
    for _, conn in pairs(self.Connections) do
        if conn then conn:Disconnect() end
    end
    self.Connections = {}
    
    if self.LoopConnection then
        self.LoopConnection:Disconnect()
        self.LoopConnection = nil
    end
    
    self.Cache = {} -- Clear cache
end

function ItemESP:Create(item)
    if self.Objects[item] then return end
    if not item.Parent then return end
    
    local char = item
    if not char:FindFirstChildOfClass("Humanoid") then return end
    
    -- ‡∏™‡∏£‡πâ‡∏≤‡∏á Highlight
    local hl = Instance.new("Highlight")
    hl.Name = "XESPHighlight"
    hl.Adornee = char
    hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    hl.FillColor = item.Team and item.Team.TeamColor.Color or Color3.fromRGB(255, 80, 80)
    hl.FillTransparency = self.Transparency
    hl.OutlineColor = Color3.fromRGB(255, 255, 255)
    hl.OutlineTransparency = 0.2
    
    pcall(function()
        hl.Parent = game:GetService("CoreGui")
    end)
    if not hl.Parent then
        hl.Parent = workspace.CurrentCamera
    end
    
    -- ‡∏™‡∏£‡πâ‡∏≤‡∏á Billboard
    local bb = Instance.new("BillboardGui")
    bb.Name = "XItemLabel"
    bb.Size = UDim2.new(0, 200, 0, 40)
    bb.StudsOffset = Vector3.new(0, 3, 0)
    bb.AlwaysOnTop = true
    bb.MaxDistance = self.Range
    bb.Adornee = char:FindFirstChild("HumanoidRootPart") or char.PrimaryPart
    
    local txt = Instance.new("TextLabel", bb)
    txt.Size = UDim2.new(1, 0, 1, 0)
    txt.BackgroundTransparency = 1
    txt.Text = char.Name
    txt.TextColor3 = Color3.fromRGB(255, 255, 200)
    txt.TextStrokeTransparency = 0.8
    txt.Font = Enum.Font.GothamBold
    txt.TextSize = 14
    
    pcall(function()
        bb.Parent = game:GetService("CoreGui")
    end)
    if not bb.Parent then
        bb.Parent = workspace.CurrentCamera
    end
    
    self.Objects[item] = {
        Highlight = hl,
        Billboard = bb,
        TextLabel = txt,
        LastUpdate = os.clock()
    }
end

function ItemESP:Update()
    if not self.Enabled then return end
    
    -- ‡∏≠‡∏±‡∏û‡πÄ‡∏î‡∏ó‡∏ó‡∏∏‡∏Å‡πÄ‡∏ü‡∏£‡∏°‡πÅ‡∏ï‡πà Optimize
    if os.clock() - self.LastUpdate < 0.1 then return end -- 10 FPS
    self.LastUpdate = os.clock()
    
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    
    local hrp = char.HumanoidRootPart
    local charPos = hrp.Position
    
    -- ‡∏™‡πÅ‡∏Å‡∏ô‡πÅ‡∏ö‡∏ö Batch ‡∏ó‡∏∏‡∏Å 2 ‡∏ß‡∏¥‡∏ô‡∏≤‡∏ó‡∏µ
    if os.clock() - self.LastUpdate > self.ScanInterval then
        self:ScanAndCreate()
    end
    
    -- ‡∏≠‡∏±‡∏û‡πÄ‡∏î‡∏ó‡∏£‡∏∞‡∏¢‡∏∞‡∏ó‡∏≤‡∏á
    for item, data in pairs(self.Objects) do
        if not item.Parent or not data.Highlight or not data.Highlight.Parent then
            if data.Highlight then data.Highlight:Destroy() end
            if data.Billboard then data.Billboard:Destroy() end
            self.Objects[item] = nil
        else
            local primary = item.PrimaryPart or item:FindFirstChildWhichIsA("BasePart")
            if primary then
                local dist = (primary.Position - charPos).Magnitude
                if dist > self.Range then
                    data.Billboard.Enabled = false
                    data.Highlight.Enabled = false
                else
                    data.Billboard.Enabled = true
                    data.Highlight.Enabled = true
                    data.TextLabel.Text = string.format("%s\n[%.0fm]", item.Name, dist)
                end
            end
        end
    end
end

function ItemESP:ScanAndCreate()
    local count = 0
    local charPos = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and LocalPlayer.Character.HumanoidRootPart.Position
    
    if not charPos then return end
    
    for _, obj in ipairs(workspace:GetChildren()) do
        if count >= self.MaxItems then break end
        
        if not self.Objects[obj] and self:IsItem(obj) then
            local primary = obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart")
            if primary then
                local dist = (primary.Position - charPos).Magnitude
                if dist <= self.Range then
                    self:Create(obj)
                    count = count + 1
                end
            end
        end
    end
end

function ItemESP:Toggle()
    self.Enabled = not self.Enabled
    
    if self.Enabled then
        print("[ItemESP] ‡πÄ‡∏õ‡∏¥‡∏î‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô (Optimized)")
        self:Clear()
        self:ScanAndCreate()
        
        self.LoopConnection = game:GetService("RunService").Heartbeat:Connect(function()
            self:Update()
        end)
        
        self.Connections.ChildAdded = workspace.ChildAdded:Connect(function(child)
            task.wait(0.5)
            if self.Enabled and self:IsItem(child) then
                self:Create(child)
            end
        end)
    else
        print("[ItemESP] ‡∏õ‡∏¥‡∏î‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô")
        self:Clear()
    end
    
    return self.Enabled
end

local Xray = {
    Enabled = false,
    Objects = {}, -- [Part] = OriginalTransparency
    Connection = nil,
    Range = 100
}

function Xray:Scan()
    if not self.Enabled then return end
    if not LocalPlayer.Character then return end
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    
    local pos = hrp.Position
    local processed = {}
    
    -- ‡∏´‡∏≤ Parts ‡πÉ‡∏ô‡∏£‡∏±‡∏®‡∏°‡∏µ (‡πÉ‡∏ä‡πâ FindPartsInBall ‡πÄ‡∏£‡πá‡∏ß‡∏Å‡∏ß‡πà‡∏≤)
    local parts = workspace:GetPartBoundsInRadius(pos, self.Range)
    
    for _, part in ipairs(parts) do
        if part.Anchored and part.Transparency < 0.9 then
            -- ‡∏Ç‡πâ‡∏≤‡∏°‡∏ï‡∏±‡∏ß‡∏•‡∏∞‡∏Ñ‡∏£, Terrain, ‡∏ô‡πâ‡∏≥
            if part:IsDescendantOf(LocalPlayer.Character) then continue end
            if part.Parent and part.Parent:FindFirstChild("Humanoid") then continue end
            if part:IsA("Terrain") or part.Name:lower():find("terrain") then continue end
            
            processed[part] = true

            part.Transparency = 0.75
        end
    end
    
    -- ‡∏Ñ‡∏∑‡∏ô‡∏Ñ‡πà‡∏≤‡∏™‡πà‡∏ß‡∏ô‡∏ó‡∏µ‡πà‡∏≠‡∏≠‡∏Å‡∏à‡∏≤‡∏Å‡∏£‡∏∞‡∏¢‡∏∞

        end
    end

-- ‡∏´‡∏≤‡πÄ‡∏õ‡πâ‡∏≤‡∏´‡∏°‡∏≤‡∏¢‡∏ó‡∏µ‡πà‡∏î‡∏µ‡∏ó‡∏µ‡πà‡∏™‡∏∏‡∏î
function Aimbot:FindTarget()
    local camera = workspace.CurrentCamera
    local center = Vector2.new(camera.ViewportSize.X/2, camera.ViewportSize.Y/2)
    local char = LocalPlayer.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    
    if not root then return nil end
    
    local targets = {}
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local pChar = player.Character
            local humanoid = pChar:FindFirstChildOfClass("Humanoid")
            
            -- Team Check
            if LocalPlayer.Team and player.Team and LocalPlayer.Team == player.Team then
                continue
            end
            
            if humanoid and humanoid.Health > 0 then
                local targetPart = pChar:FindFirstChild(self.Settings.TargetPart) 
                                  or pChar:FindFirstChild("HumanoidRootPart")
                                  or pChar:FindFirstChild("Head")
                
                if targetPart then
                    local pos, onScreen = camera:WorldToViewportPoint(targetPart.Position)
                    
                    if onScreen then
                        local screenPos = Vector2.new(pos.X, pos.Y)
                        local screenDist = (screenPos - center).Magnitude
                        local worldDist = (root.Position - targetPart.Position).Magnitude
                        
                        -- Wall Check
                        if self.Settings.WallCheck then
                            if not self:CanSee(targetPart) then
                                continue
                            end
                        end
                        
                        -- ‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì‡∏Ñ‡∏∞‡πÅ‡∏ô‡∏ô (Hybrid Score)
                        local score = 0
                        if self.AimType == "FOV" then
                            score = screenDist
                        elseif self.AimType == "Distance" then
                            score = worldDist
                        else -- Hybrid
                            score = screenDist * 0.6 + worldDist * 0.4
                        end
                        
                        table.insert(targets, {
                            Player = player,
                            Part = targetPart,
                            ScreenPos = screenPos,
                            ScreenDist = screenDist,
                            WorldDist = worldDist,
                            Score = score,
                            Velocity = targetPart.Velocity
                        })
                    end
                end
            end
        end
    end
    
    if #targets == 0 then return nil end
    
    -- ‡πÄ‡∏£‡∏µ‡∏¢‡∏á‡∏ï‡∏≤‡∏°‡∏Ñ‡∏∞‡πÅ‡∏ô‡∏ô‡∏ï‡πà‡∏≥‡∏™‡∏∏‡∏î (‡∏î‡∏µ‡∏ó‡∏µ‡πà‡∏™‡∏∏‡∏î)
    table.sort(targets, function(a, b) return a.Score < b.Score end)
    
    -- ‡∏ñ‡πâ‡∏≤‡πÄ‡∏õ‡πá‡∏ô FOV mode ‡∏ï‡πâ‡∏≠‡∏á‡∏≠‡∏¢‡∏π‡πà‡πÉ‡∏ô‡∏ß‡∏á‡∏î‡πâ‡∏ß‡∏¢
    if self.AimType == "FOV" or self.AimType == "Hybrid" then
        if targets[1].ScreenDist > self.Settings.FOV then
            return nil
        end
    end
    
    return targets[1]
end

-- Wall Check (Raycast)
function Aimbot:CanSee(part)
    local camera = workspace.CurrentCamera
    local char = LocalPlayer.Character
    
    local rayParams = RaycastParams.new()
    rayParams.FilterDescendantsInstances = {char}
    rayParams.FilterType = Enum.RaycastFilterType.Blacklist
    
    local result = workspace:Raycast(camera.CFrame.Position, (part.Position - camera.CFrame.Position).Unit * 1000, rayParams)
    
    if result then
        return result.Instance:IsDescendantOf(part.Parent)
    end
    
    return true
end

-- Silent Aim Function (‡πÇ‡∏Ñ‡∏ï‡∏£‡πÇ‡∏´‡∏î!)
function Aimbot:SilentAim(target)

        -- ‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì‡∏Å‡∏≤‡∏£‡∏ô‡∏≥ (Velocity * Time)
        local velocity = target.Part.Velocity
        local prediction = velocity * self.Settings.PredictionVelocity
        aimPosition = aimPosition + prediction
    end
    
    -- ‡∏™‡∏£‡πâ‡∏≤‡∏á CFrame ‡πÉ‡∏´‡∏°‡πà‡∏ó‡∏µ‡πà‡∏ä‡∏µ‡πâ‡πÑ‡∏õ‡∏ó‡∏µ‡πà‡πÄ‡∏õ‡πâ‡∏≤ ‡πÅ‡∏ï‡πà‡πÑ‡∏°‡πà‡∏Ç‡∏¢‡∏±‡∏ö‡∏Å‡∏•‡πâ‡∏≠‡∏á‡∏à‡∏£‡∏¥‡∏á‡πÜ

    local camera = workspace.CurrentCamera
    local oldCFrame = camera.CFrame

    -- ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö Roblox ‡∏ó‡∏±‡πà‡∏ß‡πÑ‡∏õ ‡πÄ‡∏£‡∏≤‡∏ï‡πâ‡∏≠‡∏á hook ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏ó‡∏µ‡πà‡πÄ‡∏Å‡∏°‡πÉ‡∏ä‡πâ‡∏≠‡πà‡∏≤‡∏ô CFrame

        camera.CFrame = fakeCFrame

end

-- ‡∏£‡∏∞‡∏ö‡∏ö‡∏¢‡∏¥‡∏á‡∏≠‡∏±‡∏ï‡πÇ‡∏ô‡∏°‡∏±‡∏ï‡∏¥ (Trigger Bot)
function Aimbot:TriggerBot(target)
    if not self.Settings.TriggerBot or not target then return end
    
    -- ‡πÄ‡∏ä‡πá‡∏Ñ‡∏ß‡πà‡∏≤‡πÄ‡∏õ‡πá‡∏ô‡πÄ‡∏õ‡πâ‡∏≤‡∏ó‡∏µ‡πà‡πÄ‡∏£‡∏≤‡∏•‡πá‡∏≠‡∏Ñ‡πÑ‡∏ß‡πâ‡∏à‡∏£‡∏¥‡∏á‡πÜ
    if target.ScreenDist < 20 then -- ‡∏≠‡∏¢‡∏π‡πà‡πÉ‡∏Å‡∏•‡πâ‡∏Å‡∏∂‡πà‡∏á‡∏Å‡∏•‡∏≤‡∏á‡∏à‡∏≠‡∏û‡∏≠‡∏™‡∏°‡∏Ñ‡∏ß‡∏£
        local tool = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Tool")
        if tool then
            tool:Activate()
        end
    end
end

-- Rage Mode (Spin + Instant)
function Aimbot:RageMode(target)
    if not target or not target.Part then return end
    
    local camera = workspace.CurrentCamera
    local targetCF = CFrame.new(camera.CFrame.Position, target.Part.Position)
    
    -- ‡∏´‡∏°‡∏∏‡∏ô‡∏Å‡∏•‡πâ‡∏≠‡∏á‡∏ó‡∏±‡∏ô‡∏ó‡∏µ‡πÑ‡∏°‡πà‡∏™‡∏ô‡∏Ñ‡∏ß‡∏≤‡∏°‡∏•‡∏∑‡πà‡∏ô
    camera.CFrame = targetCF
    
    -- ‡∏¢‡∏¥‡∏á‡∏ó‡∏±‡∏ô‡∏ó‡∏µ
    local tool = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Tool")
    if tool then
        tool:Activate()
    end
end

-- Normal Mode (Smooth)
function Aimbot:NormalAim(target)
    if not target or not target.Part then return end
    
    local camera = workspace.CurrentCamera
    local targetPos = target.Part.Position
    
    -- Prediction
    if self.Settings.Prediction then
        targetPos = targetPos + (target.Part.Velocity * self.Settings.PredictionVelocity)
    end
    
    local targetCF = CFrame.new(camera.CFrame.Position, targetPos)
    camera.CFrame = camera.CFrame:Lerp(targetCF, self.Settings.Smoothness)
    
    -- Auto Fire ‡∏ñ‡πâ‡∏≤‡πÄ‡∏õ‡∏¥‡∏î TriggerBot
    self:TriggerBot(target)
end

-- Render Update (FOV Circle + Snaplines)
function Aimbot:UpdateVisuals()
    if not self.fovCircle then
        self.fovCircle = self:CreateFOV()
    end
    
    if not self.snapline then
        self.snapline = self:CreateSnapline()
    end
    
    local camera = workspace.CurrentCamera
    local center = Vector2.new(camera.ViewportSize.X/2, camera.ViewportSize.Y/2)
    
    -- Update FOV
    self.fovCircle.Visible = self.Settings.ShowFOV and self.Enabled
    self.fovCircle.Radius = self.Settings.FOV
    self.fovCircle.Position = center
    self.fovCircle.Color = self.Target and Color3.fromRGB(255, 0, 0) or self.Settings.FOVColor
    
    -- Update Snapline
    if self.Settings.Snaplines and self.Target and self.Enabled then
        self.snapline.Visible = true
        self.snapline.From = center
        self.snapline.To = self.Target.ScreenPos
        self.snapline.Color = Color3.fromRGB(0, 255, 0)
    else
        self.snapline.Visible = false
    end
end

-- Main Update
function Aimbot:Update()
    if not self.Enabled then 
        self.Target = nil
        self:UpdateVisuals()
        return 
    end
    
    -- ‡∏´‡∏≤‡πÄ‡∏õ‡πâ‡∏≤
    self.Target = self:FindTarget()
    
    -- Execute Aim
    if self.Target then
        if self.Mode == "Silent" then
            self:SilentAim(self.Target)
            self:TriggerBot(self.Target)
        elseif self.Mode == "Normal" then
            self:NormalAim(self.Target)
        elseif self.Mode == "Rage" then
            self:RageMode(self.Target)
        end
    end
    
    -- ‡∏≠‡∏±‡∏û‡πÄ‡∏î‡∏ó Visuals
    self:UpdateVisuals()
end

function Aimbot:Toggle()
    self.Enabled = not self.Enabled
    
    if self.Enabled then
        print("[Aimbot] v3.0 Enabled | Mode:", self.Mode, "| Type:", self.AimType)
        
        -- RenderStepped ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö Aim
        self.Connection = game:GetService("RunService").RenderStepped:Connect(function()
            self:Update()
        end)
    else
        print("[Aimbot] Disabled")
        if self.Connection then
            self.Connection:Disconnect()
            self.Connection = nil
        end
        
        -- Cleanup visuals
        if self.fovCircle then self.fovCircle.Visible = false end
        if self.snapline then self.snapline.Visible = false end
    end
    
    return self.Enabled
end

    if not player.Character then 
        print("[Teleport] ‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô‡πÑ‡∏°‡πà‡∏°‡∏µ‡∏ï‡∏±‡∏ß‡∏•‡∏∞‡∏Ñ‡∏£")

    local myChar = LocalPlayer.Character

function Teleport:ToClosest()
    local myHRP = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not myHRP then return end
    
    local closest = nil
    local minDist = math.huge
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local hrp = player.Character:FindFirstChild("HumanoidRootPart")
            if hrp then
                local dist = (myHRP.Position - hrp.Position).Magnitude
                if dist < minDist then
                    minDist = dist
                    closest = player
                end
            end
        end
    end
    
    if closest then
        self:To(closest)
    else
        print("[Teleport] ‡πÑ‡∏°‡πà‡∏û‡∏ö‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô")
    end
end

            if self.Enabled then

                -- ‡∏ñ‡πâ‡∏≤‡πÄ‡∏Å‡∏°‡∏û‡∏¢‡∏≤‡∏¢‡∏≤‡∏°‡πÄ‡∏õ‡∏•‡∏µ‡πà‡∏¢‡∏ô ‡πÉ‡∏´‡πâ‡∏î‡∏±‡∏ô‡∏Å‡∏•‡∏±‡∏ö

            self.Connection:Disconnect()
            self.Connection = nil
        end
        self:Restore()
    end
    
    return self.Enabled
end

function Universal:DetectSystem()
    if tick() - self.Detection.DetectedAt < 10 then return end -- ‡∏ï‡∏£‡∏ß‡∏à‡∏ó‡∏∏‡∏Å 10 ‡∏ß‡∏¥
    
    print("[Universal] ‡∏Å‡∏≥‡∏•‡∏±‡∏á‡∏ï‡∏£‡∏ß‡∏à‡∏´‡∏≤‡∏£‡∏∞‡∏ö‡∏ö‡πÄ‡∏Å‡∏°...")
    local detected = {}
    
    -- 1. ‡∏ï‡∏£‡∏ß‡∏à ClickDetector ‡πÉ‡∏ô‡∏õ‡∏∑‡∏ô
    local clickCount = 0
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("ClickDetector") then
            if obj.Parent and (obj.Parent.Name:lower():find("gun") or obj.Parent.Name:lower():find("weapon")) then
                clickCount = clickCount + 1
            end
        end
    end
    if clickCount > 0 then table.insert(detected, "ClickDetector") end
    
    -- 2. ‡∏ï‡∏£‡∏ß‡∏à Tool ‡πÉ‡∏ô Player
    local char = LocalPlayer.Character
    if char then
        for _, obj in ipairs(char:GetChildren()) do
            if obj:IsA("Tool") then
                table.insert(detected, "Tool")
                break
            end
        end
    end
    
    -- 3. ‡∏ï‡∏£‡∏ß‡∏à ProximityPrompt (‡∏ã‡∏∑‡πâ‡∏≠/‡∏´‡∏¢‡∏¥‡∏ö)
    local promptCount = 0
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") then
            if obj.ActionText and (obj.ActionText:lower():find("buy") or obj.ActionText:lower():find("get")) then
                promptCount = promptCount + 1
            end
        end
    end
    if promptCount > 0 then table.insert(detected, "ProximityPrompt") end
    
    -- 4. ‡∏ï‡∏£‡∏ß‡∏à RemoteEvent ‡∏ó‡∏µ‡πà‡∏ô‡πà‡∏≤‡∏à‡∏∞‡πÉ‡∏ä‡πâ‡∏¢‡∏¥‡∏á (‡∏î‡∏π‡∏à‡∏≤‡∏Å‡∏ä‡∏∑‡πà‡∏≠)
    for _, obj in ipairs(game.ReplicatedStorage:GetDescendants()) do
        if obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction") then
            local name = obj.Name:lower()
            if name:find("shoot") or name:find("fire") or name:find("gun") or name:find("damage") then
                table.insert(detected, "RemoteEvent")
                self.RemoteCache[obj.Name] = obj
            end
        end
    end
    
    -- 5. ‡∏ï‡∏£‡∏ß‡∏à Anti-Cheat (‡∏ß‡∏¥‡∏ò‡∏µ‡∏á‡πà‡∏≤‡∏¢: ‡∏î‡∏π‡∏ß‡πà‡∏≤‡∏°‡∏µ‡∏™‡∏Ñ‡∏£‡∏¥‡∏õ‡∏ï‡πå‡∏ä‡∏∑‡πà‡∏≠‡πÅ‡∏õ‡∏•‡∏Å‡πÜ ‡∏°‡∏±‡πâ‡∏¢)
    local acDetected = 0
    for _, obj in ipairs(game:GetService("Players").LocalPlayer.PlayerScripts:GetChildren()) do
        local name = obj.Name:lower()
        if name:find("ac") or name:find("anticheat") or name:find("detect") or name:find("security") then
            acDetected = acDetected + 1
        end
    end
    self.Detection.AntiCheatLevel = math.min(acDetected, 3)
    
    -- ‡πÄ‡∏•‡∏∑‡∏≠‡∏Å‡∏£‡∏∞‡∏ö‡∏ö‡∏´‡∏•‡∏±‡∏Å
    if table.find(detected, "ClickDetector") then
        self.Detection.SystemType = "ClickDetector"
    elseif table.find(detected, "Tool") then
        self.Detection.SystemType = "Tool"
    elseif table.find(detected, "RemoteEvent") then
        self.Detection.SystemType = "RemoteEvent"
    elseif table.find(detected, "ProximityPrompt") then
        self.Detection.SystemType = "ProximityPrompt"
    end
    
    self.Detection.DetectedAt = tick()
    
    -- ‡∏õ‡∏£‡∏±‡∏ö Settings ‡∏ï‡∏≤‡∏° Anti-Cheat
    if self.Detection.AntiCheatLevel >= 2 then
        self.Settings.SafetyDelay = 0.05

    end
    
    print(string.format("[Universal] ‡∏ï‡∏£‡∏ß‡∏à‡∏û‡∏ö: %s (AC Level: %d)", 
        self.Detection.SystemType, self.Detection.AntiCheatLevel))
    
    return self.Detection.SystemType
end

-- ‡∏¢‡∏¥‡∏á‡πÅ‡∏ö‡∏ö Universal (‡∏õ‡∏£‡∏±‡∏ö‡∏ï‡∏≤‡∏°‡πÄ‡∏Å‡∏°)
function Universal:Fire()

    if not char then return false end
    
    local fired = false
    
    -- ‡∏ß‡∏¥‡∏ò‡∏µ 1: ‡∏ï‡∏≤‡∏°‡∏£‡∏∞‡∏ö‡∏ö‡∏ó‡∏µ‡πà‡∏ï‡∏£‡∏ß‡∏à‡∏û‡∏ö
    if self.Detection.SystemType == "ClickDetector" then
        fired = self:FireClickDetector()
    elseif self.Detection.SystemType == "Tool" then
        fired = self:FireTool()
    elseif self.Detection.SystemType == "RemoteEvent" then
        fired = self:FireRemote()
    elseif self.Detection.SystemType == "ProximityPrompt" then
        fired = self:FirePrompt()
    end
    
    -- ‡∏ß‡∏¥‡∏ò‡∏µ 2: Backup (‡∏ñ‡πâ‡∏≤‡∏ß‡∏¥‡∏ò‡∏µ‡∏´‡∏•‡∏±‡∏Å‡πÑ‡∏°‡πà‡πÄ‡∏ß‡∏¥‡∏£‡πå‡∏Ñ)
    if not fired and self.Settings.BackupTool then
        fired = self:FireTool() -- ‡∏•‡∏≠‡∏á‡πÉ‡∏ä‡πâ Tool ‡πÄ‡∏™‡∏°‡∏≠‡πÄ‡∏õ‡πá‡∏ô‡∏™‡∏≥‡∏£‡∏≠‡∏á
    end
    
    -- ‡∏ß‡∏¥‡∏ò‡∏µ 3: Mouse Click (‡πÉ‡∏ä‡πâ‡πÑ‡∏î‡πâ‡πÄ‡∏â‡∏û‡∏≤‡∏∞ PC)
    if not fired and self.Settings.UseMouse then
        pcall(function()
            mouse1click()
            fired = true
        end)
    end
    
    -- ‡∏ô‡∏±‡∏ö‡∏Ñ‡∏ß‡∏≤‡∏°‡∏•‡πâ‡∏°‡πÄ‡∏´‡∏•‡∏ß (‡∏õ‡∏£‡∏±‡∏ö‡∏£‡∏∞‡∏ö‡∏ö‡∏ñ‡πâ‡∏≤‡∏¢‡∏¥‡∏á‡πÑ‡∏°‡πà‡πÑ‡∏î‡πâ‡∏´‡∏•‡∏≤‡∏¢‡∏Ñ‡∏£‡∏±‡πâ‡∏á)
    if not fired then
        self.ConsecutiveFailures = self.ConsecutiveFailures + 1
        if self.ConsecutiveFailures > 10 then
            print("[Universal] ‡∏¢‡∏¥‡∏á‡πÑ‡∏°‡πà‡∏™‡∏≥‡πÄ‡∏£‡πá‡∏à‡∏´‡∏•‡∏≤‡∏¢‡∏Ñ‡∏£‡∏±‡πâ‡∏á ‡∏Å‡∏≥‡∏•‡∏±‡∏á‡∏™‡πÅ‡∏Å‡∏ô‡πÉ‡∏´‡∏°‡πà...")
            self:DetectSystem()
            self.ConsecutiveFailures = 0
        end
    else
        self.ConsecutiveFailures = 0
    end
    
    return fired
end

function Universal:FireClickDetector()
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    
    -- ‡∏´‡∏≤‡∏õ‡∏∑‡∏ô‡πÉ‡∏Å‡∏•‡πâ‡∏™‡∏∏‡∏î
    local nearestGun = nil
    local minDist = 20 -- ‡∏£‡∏∞‡∏¢‡∏∞‡πÉ‡∏Å‡∏•‡πâ‡∏°‡∏∑‡∏≠
    
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("ClickDetector") then
            local parent = obj.Parent
            if parent and parent:IsA("BasePart") then
                if parent.Name:lower():find("gun") or parent.Name:lower():find("weapon") then
                    local dist = (hrp.Position - parent.Position).Magnitude
                    if dist < minDist then
                        minDist = dist
                        nearestGun = obj
                    end
                end
            end
        end
    end
    
    if nearestGun then
        pcall(function()
            if fireclickdetector then
                fireclickdetector(nearestGun)
            else
                nearestGun:Click()
            end
        end)
        return true
    end
    return false
end

end

-- ‡∏î‡∏µ‡πÄ‡∏•‡∏¢‡πå‡∏ó‡∏µ‡πà‡∏õ‡∏•‡∏≠‡∏î‡∏†‡∏±‡∏¢ (‡∏™‡∏∏‡πà‡∏°‡πÄ‡∏•‡πá‡∏Å‡∏ô‡πâ‡∏≠‡∏¢)
function Universal:GetSafeDelay()
    local base = 1 / self.Settings.CPS
    local safety = self.Settings.SafetyDelay
    
    -- ‡∏™‡∏∏‡πà‡∏° ¬±10% ‡πÉ‡∏´‡πâ‡∏î‡∏π‡πÄ‡∏õ‡πá‡∏ô‡∏ò‡∏£‡∏£‡∏°‡∏ä‡∏≤‡∏ï‡∏¥
    local random = math.random(90, 110) / 100
    
    return (base + safety) * random
end

function Universal:Update()
    if not self.Enabled then return end
    
    -- ‡∏ï‡∏£‡∏ß‡∏à‡∏´‡∏≤‡∏£‡∏∞‡∏ö‡∏ö‡∏ñ‡πâ‡∏≤‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡∏£‡∏π‡πâ
    if self.Detection.SystemType == "Unknown" then
        self:DetectSystem()
        task.wait(1)
    end
    
    -- ‡∏¢‡∏¥‡∏á
    local fired = self:Fire()
    
    -- ‡∏î‡∏µ‡πÄ‡∏•‡∏¢‡πå‡∏ó‡∏µ‡πà‡πÄ‡∏´‡∏°‡∏≤‡∏∞‡∏™‡∏°
    local delay = self:GetSafeDelay()
    if fired then
        task.wait(delay)
    else
        task.wait(0.1) -- ‡∏ñ‡πâ‡∏≤‡∏¢‡∏¥‡∏á‡πÑ‡∏°‡πà‡πÑ‡∏î‡πâ ‡∏£‡∏≠‡πÅ‡∏•‡πâ‡∏ß‡∏•‡∏≠‡∏á‡πÉ‡∏´‡∏°‡πà
    end
end

function Universal:Toggle()
    self.Enabled = not self.Enabled
    
    if self.Enabled then
        self:DetectSystem()
        print("[Universal] Enabled - Auto Detect Mode")
        
        -- ‡πÉ‡∏ä‡πâ Heartbeat ‡πÅ‡∏ó‡∏ô RenderStepped (‡πÄ‡∏£‡πá‡∏ß‡∏Å‡∏ß‡πà‡∏≤)
        self.Connection = game:GetService("RunService").Heartbeat:Connect(function()
            self:Update()
        end)
    else
        print("[Universal] Disabled")
        if self.Connection then
            self.Connection:Disconnect()
            self.Connection = nil
        end
    end
    return self.Enabled
end

    Range = 200,

-- ‡∏ï‡∏£‡∏ß‡∏à‡∏à‡∏±‡∏ö NPC ‡∏î‡πâ‡∏ß‡∏¢‡∏´‡∏•‡∏≤‡∏¢‡∏ß‡∏¥‡∏ò‡∏µ (‡∏ä‡∏∑‡πà‡∏≠, Humanoid, ‡πÑ‡∏°‡πà‡πÉ‡∏ä‡πà‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô)
function NPCWarp:IsNPC(obj)
    if not obj:IsA("Model") then return false end
    if obj.Parent == nil then return false end
    
    -- ‡πÑ‡∏°‡πà‡πÉ‡∏ä‡πà‡∏ï‡∏±‡∏ß‡∏•‡∏∞‡∏Ñ‡∏£‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô
    if game.Players:GetPlayerFromCharacter(obj) then return false end
    
    -- ‡πÄ‡∏ä‡πá‡∏Ñ‡∏ä‡∏∑‡πà‡∏≠ NPC
    local name = obj.Name:lower()
    local patterns = {"npc", "bot", "mob", "enemy", "monster", "villager", "quest", "civilian", "guard", "shop"}
    for _, pattern in ipairs(patterns) do
        if name:find(pattern) then
            return true
        end
    end
    
    -- ‡πÄ‡∏ä‡πá‡∏Ñ Humanoid ‡πÅ‡∏ï‡πà‡πÑ‡∏°‡πà‡πÉ‡∏ä‡πà‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô
    if obj:FindFirstChildOfClass("Humanoid") then
        local hum = obj:FindFirstChildOfClass("Humanoid")
        -- ‡∏ï‡πâ‡∏≠‡∏á‡∏°‡∏µ PrimaryPart ‡∏î‡πâ‡∏ß‡∏¢
        if obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart") then
            return true
        end
    end
    
    return false
end

-- ‡∏™‡πÅ‡∏Å‡∏ô‡∏´‡∏≤ NPC ‡∏ó‡∏±‡πâ‡∏á‡∏´‡∏°‡∏î‡πÉ‡∏ô‡∏û‡∏∑‡πâ‡∏ô‡∏ó‡∏µ‡πà
function NPCWarp:ScanAll()
    if tick() - self.LastScan < 2 then return self.FoundNPCs end -- Cache 2 ‡∏ß‡∏¥‡∏ô‡∏≤‡∏ó‡∏µ
    
    self.FoundNPCs = {}
    local myHRP = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not myHRP then return {} end
    
    for _, obj in ipairs(workspace:GetChildren()) do
        if self:IsNPC(obj) then
            local part = obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart")
            if part then
                local dist = (myHRP.Position - part.Position).Magnitude
                if dist <= self.Range then
                    table.insert(self.FoundNPCs, {
                        Model = obj,
                        Part = part,
                        Distance = dist,
                        Name = obj.Name
                    })
                end
            end
        end
    end
    
    -- ‡πÄ‡∏£‡∏µ‡∏¢‡∏á‡∏ï‡∏≤‡∏°‡∏£‡∏∞‡∏¢‡∏∞‡∏ó‡∏≤‡∏á (‡πÉ‡∏Å‡∏•‡πâ‡∏™‡∏∏‡∏î‡∏Å‡πà‡∏≠‡∏ô)
    table.sort(self.FoundNPCs, function(a,b) return a.Distance < b.Distance end)
    
    self.LastScan = tick()
    return self.FoundNPCs
end

-- ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡πÄ‡∏î‡∏¥‡∏ô/‡∏ö‡∏¥‡∏ô‡πÑ‡∏õ‡∏´‡∏≤ NPC ‡πÅ‡∏ö‡∏ö‡∏≠‡∏±‡∏ï‡πÇ‡∏ô‡∏°‡∏±‡∏ï‡∏¥
function NPCWarp:AutoMoveTo(targetData)
    if not targetData then return end
    if not LocalPlayer.Character then return end
    
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    local hum = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
    if not hrp or not hum then return end
    
    local targetPos = targetData.Part.Position
    local dist = targetData.Distance
    
    -- ‡∏ñ‡πâ‡∏≤‡πÑ‡∏Å‡∏•‡∏Å‡∏ß‡πà‡∏≤ 10 studs ‡πÉ‡∏´‡πâ‡πÄ‡∏Ñ‡∏•‡∏∑‡πà‡∏≠‡∏ô‡∏ó‡∏µ‡πà
    if dist > 10 then
        -- ‡πÄ‡∏î‡∏¥‡∏ô‡πÅ‡∏ö‡∏ö‡∏•‡∏∑‡πà‡∏ô‡πÑ‡∏´‡∏•‡∏î‡πâ‡∏ß‡∏¢ Tween
        local direction = (targetPos - hrp.Position).Unit
        local movePos = targetPos - (direction * 5) -- ‡∏´‡∏¢‡∏∏‡∏î‡∏´‡πà‡∏≤‡∏á 5 studs
        
        -- ‡πÉ‡∏ä‡πâ BodyPosition ‡∏´‡∏£‡∏∑‡∏≠ Lerp ‡πÉ‡∏´‡πâ‡∏Å‡∏≤‡∏£‡πÄ‡∏Ñ‡∏•‡∏∑‡πà‡∏≠‡∏ô‡∏ó‡∏µ‡πà‡∏•‡∏∑‡πà‡∏ô
        hrp.CFrame = hrp.CFrame:Lerp(CFrame.new(movePos + Vector3.new(0, 2, 0)), 0.1)
        hrp.Velocity = Vector3.new(0, 0, 0)
        
        -- ‡∏ö‡∏±‡∏á‡∏Ñ‡∏±‡∏ö‡πÉ‡∏´‡πâ‡∏Å‡∏£‡∏∞‡πÇ‡∏î‡∏î‡∏Ç‡πâ‡∏≤‡∏°‡∏™‡∏¥‡πà‡∏á‡∏Å‡∏µ‡∏î‡∏Ç‡∏ß‡∏≤‡∏á
        if math.abs(hrp.Position.Y - targetPos.Y) > 5 then
            hum:ChangeState(Enum.HumanoidStateType.Jumping)
        end
    else
        -- ‡∏≠‡∏¢‡∏π‡πà‡πÉ‡∏Å‡∏•‡πâ‡πÅ‡∏•‡πâ‡∏ß ‡∏ó‡∏≥ Action
        self:PerformAction(targetData.Model)
    end
end

-- ‡∏ó‡∏≥ Action ‡∏Å‡∏±‡∏ö NPC (Interact ‡∏´‡∏£‡∏∑‡∏≠ Attack)
function NPCWarp:PerformAction(npc)
    if not npc then return end
    
    if self.Mode == "Interact" then
        -- ‡∏Å‡∏î ProximityPrompt
        local prompt = npc:FindFirstChildOfClass("ProximityPrompt")
        if prompt then
            pcall(function()
                prompt:InputHoldBegin()
                task.wait(0.2)
                prompt:InputHoldEnd()
            end)
        end
        
        -- ‡∏Å‡∏î ClickDetector (PC)
        local click = npc:FindFirstChildOfClass("ClickDetector")
        if click and fireclickdetector then
            fireclickdetector(click)

            local npcs = self:ScanAll()
            
            if #npcs > 0 then

local Hitbox = {
    Enabled = false,
    Size = 3, -- ‡∏Ç‡∏ô‡∏≤‡∏î‡∏Ñ‡∏π‡∏ì (3 = 3x)
    Connection = nil,
    LastScan = 0,
    Originals = {} -- [Part] = {Size=..., Trans=..., Mat=...}
}

function Hitbox:ProcessCharacter(char)
    if not char then return end
    local isEnemy = false
    
    -- ‡πÄ‡∏ä‡πá‡∏Ñ‡∏ß‡πà‡∏≤‡πÄ‡∏õ‡πá‡∏ô‡∏®‡∏±‡∏ï‡∏£‡∏π (‡πÑ‡∏°‡πà‡πÉ‡∏ä‡πà‡∏ó‡∏µ‡∏°‡πÄ‡∏î‡∏µ‡∏¢‡∏ß‡∏Å‡∏±‡∏ô)
    for _, player in ipairs(Players:GetPlayers()) do
        if player.Character == char and player ~= LocalPlayer then
            if LocalPlayer.Team and player.Team and LocalPlayer.Team ~= player.Team then
                isEnemy = true
            elseif not LocalPlayer.Team then
                isEnemy = true
            end
            break
        end
    end
    
    if not isEnemy then return end
    
    -- ‡∏Ç‡∏¢‡∏≤‡∏¢‡∏™‡πà‡∏ß‡∏ô‡∏™‡∏≥‡∏Ñ‡∏±‡∏ç
    local targets = {"Head", "HumanoidRootPart", "UpperTorso", "Torso"}
    for _, name in ipairs(targets) do
        local part = char:FindFirstChild(name)
        if part and part:IsA("BasePart") then
            if not self.Originals[part] then
                self.Originals[part] = {
                    Size = part.Size,
                    Transparency = part.Transparency,
                    Material = part.Material
                }
            end
            
            if self.Enabled then
                -- ‡∏Ç‡∏¢‡∏≤‡∏¢‡πÅ‡∏•‡∏∞‡∏ó‡∏≥‡πÉ‡∏´‡πâ‡πÄ‡∏´‡πá‡∏ô‡πÑ‡∏î‡πâ‡∏ä‡∏±‡∏î
                part.Size = self.Originals[part].Size * self.Size
                part.Transparency = 0.7
                part.Material = Enum.Material.Neon
            else
                -- ‡∏Ñ‡∏∑‡∏ô‡∏Ñ‡πà‡∏≤
                part.Size = self.Originals[part].Size
                part.Transparency = self.Originals[part].Transparency
                part.Material = self.Originals[part].Material
            end
        end
    end
end

function Hitbox:Scan()
    -- ‡∏™‡πÅ‡∏Å‡∏ô‡∏ó‡∏∏‡∏Å 0.5 ‡∏ß‡∏¥ (‡∏õ‡∏£‡∏∞‡∏´‡∏¢‡∏±‡∏î FPS)
    if tick() - self.LastScan < 0.5 then return end
    self.LastScan = tick()
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            self:ProcessCharacter(player.Character)
        end
    end
end

function Hitbox:Toggle()
    self.Enabled = not self.Enabled
    
    if self.Enabled then
        print("[Hitbox] ‡πÄ‡∏õ‡∏¥‡∏î‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô (x" .. self.Size .. ")")
        -- ‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏•‡∏ó‡∏±‡∏ô‡∏ó‡∏µ
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= LocalPlayer and player.Character then
                self:ProcessCharacter(player.Character)
            end
        end
        
        -- Loop ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô‡πÉ‡∏´‡∏°‡πà
        self.Connection = game:GetService("RunService").Heartbeat:Connect(function()
            self:Scan()
        end)
    else
        print("[Hitbox] ‡∏õ‡∏¥‡∏î‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô - ‡∏Ñ‡∏∑‡∏ô‡∏Ñ‡πà‡∏≤")
        if self.Connection then
            self.Connection:Disconnect()
            self.Connection = nil
        end
        
        -- ‡∏Ñ‡∏∑‡∏ô‡∏Ñ‡πà‡∏≤‡∏ó‡∏±‡πâ‡∏á‡∏´‡∏°‡∏î
        for part, data in pairs(self.Originals) do
            if part and part.Parent then
                part.Size = data.Size
                part.Transparency = data.Transparency
                part.Material = data.Material
            end
        end
        self.Originals = {}
    end
    
    return self.Enabled
end

function FlySystem.NPCFollow:FindNearest()
    if not LocalPlayer.Character then return nil end
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end
    
    local nearest = nil
    local nearestDist = math.huge
    local nearestPart = nil
    
    -- ‡∏ß‡∏ô‡∏´‡∏≤ NPC ‡πÉ‡∏ô‡∏ß‡∏¥‡∏ò‡∏µ‡∏ó‡∏µ‡πà‡∏´‡∏•‡∏≤‡∏Å‡∏´‡∏•‡∏≤‡∏¢
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("Model") and obj ~= LocalPlayer.Character then
            local isNPC = false
            local name = obj.Name:lower()
            
            -- ‡πÄ‡∏ä‡πá‡∏Ñ‡∏ä‡∏∑‡πà‡∏≠ NPC
            if name:find("npc") or name:find("bot") or name:find("mob") or 
               name:find("enemy") or name:find("villager") or name:find("quest") then
                isNPC = true
            end
            
            -- ‡πÄ‡∏ä‡πá‡∏Ñ Humanoid ‡πÅ‡∏ï‡πà‡πÑ‡∏°‡πà‡πÉ‡∏ä‡πà‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô
            if obj:FindFirstChildOfClass("Humanoid") then
                local hum = obj:FindFirstChildOfClass("Humanoid")
                if hum and hum.Health > 0 and not game.Players:GetPlayerFromCharacter(obj) then
                    isNPC = true
                end
            end
            
            if isNPC then
                local part = obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart")
                if part then
                    local dist = (hrp.Position - part.Position).Magnitude
                    if dist < self.Range and dist < nearestDist then
                        nearestDist = dist
                        nearest = obj
                        nearestPart = part
                    end
                end
            end
        end
    end
    
    return nearest, nearestPart
end

-- ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏ô‡πÄ‡∏Ñ‡∏•‡∏∑‡πà‡∏≠‡∏ô‡∏ó‡∏µ‡πà‡πÑ‡∏õ‡∏´‡∏≤ NPC (‡πÄ‡∏î‡∏¥‡∏ô‡∏´‡∏£‡∏∑‡∏≠‡∏ö‡∏¥‡∏ô)
function FlySystem.NPCFollow:MoveToTarget(npc, part)
    if not npc or not part then return end
    if not LocalPlayer.Character then return end
    
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    
    local dist = (hrp.Position - part.Position).Magnitude
    
    -- ‡∏ñ‡πâ‡∏≤‡πÑ‡∏Å‡∏•‡πÄ‡∏Å‡∏¥‡∏ô 10 studs ‡πÉ‡∏´‡πâ‡πÄ‡∏Ñ‡∏•‡∏∑‡πà‡∏≠‡∏ô‡∏ó‡∏µ‡πà‡πÑ‡∏õ
    if dist > 10 then
        -- ‡πÉ‡∏ä‡πâ Tween ‡∏´‡∏£‡∏∑‡∏≠ Lerp ‡πÉ‡∏´‡πâ‡∏Å‡∏≤‡∏£‡πÄ‡∏Ñ‡∏•‡∏∑‡πà‡∏≠‡∏ô‡∏ó‡∏µ‡πà‡∏•‡∏∑‡πà‡∏ô
        local targetPos = part.Position + Vector3.new(0, 2, 0)
        hrp.CFrame = hrp.CFrame:Lerp(CFrame.new(targetPos), 0.1)
        hrp.Velocity = Vector3.new(0, 0, 0)
    else
        -- ‡∏≠‡∏¢‡∏π‡πà‡πÉ‡∏Å‡∏•‡πâ‡πÅ‡∏•‡πâ‡∏ß ‡∏ó‡∏≥ Action
        if self.Action == "Interact" then

            local prompt = npc:FindFirstChildOfClass("ProximityPrompt")
            if prompt then
                pcall(function()
                    prompt:InputHoldBegin()
                    task.wait(0.2)
                    prompt:InputHoldEnd()
                end)
            end
        elseif self.Action == "Attack" then
            -- ‡πÉ‡∏ä‡πâ Tool ‡∏ï‡∏µ
            local tool = LocalPlayer.Character:FindFirstChildOfClass("Tool")
            if tool then
                tool:Activate()
            end
        end
    end
end

            self.Connection:Disconnect()

end

    },

    PurchasedItems = {} -- ‡∏õ‡∏£‡∏∞‡∏ß‡∏±‡∏ï‡∏¥‡∏Å‡∏≤‡∏£‡∏ã‡∏∑‡πâ‡∏≠

-- ‡∏ï‡∏£‡∏ß‡∏à‡∏à‡∏±‡∏ö‡∏Ç‡∏≠‡∏á‡∏ó‡∏µ‡πà‡∏ã‡∏∑‡πâ‡∏≠/‡∏´‡∏¢‡∏¥‡∏ö‡πÑ‡∏î‡πâ
function ItemPurchaser:IsPurchaseableItem(obj)
    if not obj then return false end
    
    -- ‡∏õ‡∏£‡∏∞‡πÄ‡∏†‡∏ó 1: ProximityPrompt (‡∏™‡∏µ‡πÄ‡∏Ç‡∏µ‡∏¢‡∏ß‡πÜ ‡πÉ‡∏´‡πâ‡∏Å‡∏î‡∏ã‡∏∑‡πâ‡∏≠)
    if obj:IsA("ProximityPrompt") then
        -- ‡∏ï‡πâ‡∏≠‡∏á‡∏°‡∏µ ObjectText ‡∏ó‡∏µ‡πà‡∏ö‡πà‡∏á‡∏ö‡∏≠‡∏Å‡∏ß‡πà‡∏≤‡πÄ‡∏õ‡πá‡∏ô‡∏Ç‡∏≠‡∏á
        if obj.ObjectText and obj.ObjectText ~= "" then
            return true, obj.ObjectText
        end
        -- ‡∏´‡∏£‡∏∑‡∏≠ ActionText ‡πÄ‡∏õ‡πá‡∏ô "Purchase", "Buy", "Collect"
        local action = obj.ActionText:lower()
        if action:find("purchase") or action:find("buy") or action:find("collect") or action:find("equip") then
            return true, obj.Parent.Name
        end
    end
    
    -- ‡∏õ‡∏£‡∏∞‡πÄ‡∏†‡∏ó 2: ClickDetector (‡∏Å‡∏î‡∏ã‡∏∑‡πâ‡∏≠)
    if obj:IsA("ClickDetector") then
        local parent = obj.Parent
        -- ‡∏ñ‡πâ‡∏≤ Parent ‡∏°‡∏µ SurfaceGui ‡∏´‡∏£‡∏∑‡∏≠ BillboardGui ‡πÅ‡∏™‡∏î‡∏á‡∏ä‡∏∑‡πà‡∏≠‡∏£‡∏≤‡∏Ñ‡∏≤
        if parent:FindFirstChildWhichIsA("SurfaceGui") or parent:FindFirstChildWhichIsA("BillboardGui") then
            return true, parent.Name
        end
        -- ‡∏´‡∏£‡∏∑‡∏≠‡∏°‡∏µ Tool ‡∏≠‡∏¢‡∏π‡πà‡πÉ‡∏ô Parent
        if parent:FindFirstChildWhichIsA("Tool") then
            return true, parent:FindFirstChildWhichIsA("Tool").Name
        end
    end
    
    -- ‡∏õ‡∏£‡∏∞‡πÄ‡∏†‡∏ó 3: Tool ‡∏ß‡∏≤‡∏á‡∏≠‡∏¢‡∏π‡πà‡πÉ‡∏ô Workspace
    if obj:IsA("Tool") and obj.Parent == workspace then
        return true, obj.Name
    end
    
    -- ‡∏õ‡∏£‡∏∞‡πÄ‡∏†‡∏ó 4: Model ‡∏ó‡∏µ‡πà‡∏°‡∏µ ProximityPrompt ‡∏≠‡∏¢‡∏π‡πà‡∏Ç‡πâ‡∏≤‡∏á‡πÉ‡∏ô
    if obj:IsA("Model") then
        local prompt = obj:FindFirstChildWhichIsA("ProximityPrompt")
        if prompt then
            return self:IsPurchaseableItem(prompt)
        end
        local click = obj:FindFirstChildWhichIsA("ClickDetector")
        if click then
            return self:IsPurchaseableItem(click)
        end
    end
    
    return false
end

-- ‡∏™‡πÅ‡∏Å‡∏ô‡∏´‡∏≤‡∏Ç‡∏≠‡∏á‡∏ó‡∏µ‡πà‡∏ã‡∏∑‡πâ‡∏≠‡πÑ‡∏î‡πâ‡πÉ‡∏ô‡πÅ‡∏°‡∏û
function ItemPurchaser:ScanItems()
    if tick() - self.LastScan < 2 then return end -- ‡∏™‡πÅ‡∏Å‡∏ô‡∏ó‡∏∏‡∏Å 2 ‡∏ß‡∏¥
    
    self.Items = {}
    local myHRP = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not myHRP then return end
    
    local charPos = myHRP.Position
    
    -- ‡∏™‡πÅ‡∏Å‡∏ô ProximityPrompt ‡πÉ‡∏ô‡∏ß‡∏á‡∏Å‡∏ß‡πâ‡∏≤‡∏á
    for _, prompt in ipairs(workspace:GetDescendants()) do
        if prompt:IsA("ProximityPrompt") then
            local parent = prompt.Parent
            if parent and parent:IsA("BasePart") then
                local dist = (parent.Position - charPos).Magnitude
                if dist <= self.Settings.ScanRadius then
                    local isItem, itemName = self:IsPurchaseableItem(prompt)
                    if isItem and itemName then
                        table.insert(self.Items, {
                            Object = prompt,
                            Name = itemName,
                            Type = "ProximityPrompt",
                            Distance = dist,
                            Position = parent.Position,
                            Price = self:GetPrice(prompt)
                        })
                    end
                end
            end
        end
    end
    
    -- ‡∏™‡πÅ‡∏Å‡∏ô ClickDetector
    for _, click in ipairs(workspace:GetDescendants()) do
        if click:IsA("ClickDetector") then
            local parent = click.Parent
            if parent and parent:IsA("BasePart") then
                local dist = (parent.Position - charPos).Magnitude
                if dist <= self.Settings.ScanRadius then
                    local isItem, itemName = self:IsPurchaseableItem(click)
                    if isItem and itemName then
                        table.insert(self.Items, {
                            Object = click,
                            Name = itemName,
                            Type = "ClickDetector",
                            Distance = dist,
                            Position = parent.Position,
                            Price = self:GetPrice(click)
                        })
                    end
                end
            end
        end
    end
    
    -- ‡∏™‡πÅ‡∏Å‡∏ô Tool ‡πÉ‡∏ô Workspace
    for _, tool in ipairs(workspace:GetChildren()) do
        if tool:IsA("Tool") then
            local handle = tool:FindFirstChild("Handle")
            if handle then
                local dist = (handle.Position - charPos).Magnitude
                if dist <= self.Settings.ScanRadius then
                    table.insert(self.Items, {
                        Object = tool,
                        Name = tool.Name,
                        Type = "Tool",
                        Distance = dist,
                        Position = handle.Position,
                        Price = "Pickup"
                    })
                end
            end
        end
    end
    
    -- ‡πÄ‡∏£‡∏µ‡∏¢‡∏á‡∏ï‡∏≤‡∏°‡∏£‡∏∞‡∏¢‡∏∞‡∏ó‡∏≤‡∏á
    table.sort(self.Items, function(a,b) return a.Distance < b.Distance end)
    
    self.LastScan = tick()
    self:UpdateUIList()
end

-- ‡∏î‡∏∂‡∏á‡∏£‡∏≤‡∏Ñ‡∏≤‡∏à‡∏≤‡∏Å UI
function ItemPurchaser:GetPrice(obj)
    local parent = obj.Parent
    
    -- ‡∏´‡∏≤ TextLabel ‡∏ó‡∏µ‡πà‡∏ô‡πà‡∏≤‡∏à‡∏∞‡πÄ‡∏õ‡πá‡∏ô‡∏£‡∏≤‡∏Ñ‡∏≤
    for _, gui in ipairs(parent:GetDescendants()) do
        if gui:IsA("TextLabel") or gui:IsA("TextButton") then
            local text = gui.Text:lower()
            if text:find("%$") or text:find("price") or text:find("cost") or text:find("%d") then
                return gui.Text
            end
        end
    end
    
    return "?"
end

-- ‡∏ã‡∏∑‡πâ‡∏≠/‡∏´‡∏¢‡∏¥‡∏ö‡πÑ‡∏≠‡πÄ‡∏ó‡πá‡∏°
function ItemPurchaser:PurchaseItem(itemData, amount)
    if not itemData or not itemData.Object then return false end
    
    local success = false
    
    for i = 1, amount do
        if itemData.Type == "ProximityPrompt" then
            success = pcall(function()
                -- ‡∏¢‡πâ‡∏≤‡∏¢‡∏ï‡∏±‡∏ß‡∏•‡∏∞‡∏Ñ‡∏£‡πÑ‡∏õ‡πÉ‡∏Å‡∏•‡πâ‡πÜ
                if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                    local hrp = LocalPlayer.Character.HumanoidRootPart
                    hrp.CFrame = CFrame.new(itemData.Position + Vector3.new(0, 5, 0))
                    task.wait(0.2)
                end
                
                -- ‡∏Å‡∏î Prompt
                local prompt = itemData.Object
                prompt:InputHoldBegin()
                task.wait(0.3)
                prompt:InputHoldEnd()
            end)
            
        elseif itemData.Type == "ClickDetector" then
            success = pcall(function()
                -- ‡∏¢‡πâ‡∏≤‡∏¢‡πÑ‡∏õ‡πÉ‡∏Å‡∏•‡πâ‡πÜ
                if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                    local hrp = LocalPlayer.Character.HumanoidRootPart
                    hrp.CFrame = CFrame.new(itemData.Position + Vector3.new(0, 5, 0))
                    task.wait(0.2)
                end
                
                -- ‡∏Å‡∏î ClickDetector
                if fireclickdetector then
                    fireclickdetector(itemData.Object)
                else
                    itemData.Object:DetectClick()
                end
            end)
            
        elseif itemData.Type == "Tool" then
            success = pcall(function()
                local tool = itemData.Object
                tool.Parent = LocalPlayer.Backpack or LocalPlayer.Character
            end)
        end
        
        if success then
            table.insert(self.PurchasedItems, {
                Name = itemData.Name,
                Time = tick()
            })
        end
        
        task.wait(0.1) -- ‡∏´‡∏ô‡πà‡∏ß‡∏á‡∏£‡∏∞‡∏´‡∏ß‡πà‡∏≤‡∏á‡∏ã‡∏∑‡πâ‡∏≠‡∏ã‡πâ‡∏≠‡∏ô
    end
    
    return success
end

-- ‡∏ã‡∏∑‡πâ‡∏≠‡∏ã‡πâ‡∏≠‡∏ô‡∏Å‡∏±‡∏ô‡∏´‡∏•‡∏≤‡∏¢‡∏Ñ‡∏£‡∏±‡πâ‡∏á
function ItemPurchaser:PurchaseMultiple(itemData, count)
    count = math.min(count, 99)
    local purchased = 0
    
    for i = 1, count do
        if self:PurchaseItem(itemData, 1) then
            purchased = purchased + 1
        else
            break
        end
    end
    
    return purchased
end
